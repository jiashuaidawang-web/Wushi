二: 服务器启动
在本机
cd /Users/null/IdeaProjects/github/Wushi
# 打成 tar.gz (排除 target/.git 等不需要的)
tar -czf /tmp/wushi.tar.gz \
--exclude='.git' \
--exclude='*/target' \
--exclude='wushi-web/node_modules' \
.


服务器上：
cd /opt/Wushi
docker compose down -v
tar -xzf /tmp/wushi.tar.gz --overwrite
docker compose up -d




cd /opt/Wushi

# 停旧服务 + 清数据卷
docker compose down -v

# 解压覆盖 (保留现有 docker-compose.yml 以外的文件)
tar -xzf /tmp/wushi.tar.gz --overwrite

# 确认修复到位
head -1 /opt/Wushi/docker/mysql-init/001_rule_config.sql
grep -c "DELIMITER" /opt/Wushi/docker/mysql-init/003_experience_growth.sql

# 重新启动
docker compose up -d

# 看 MySQL 能不能起来
docker logs wushi-mysql 2>&1 | tail -10



# 传到服务器
scp /tmp/wushi.tar.gz root@124.223.220.245:/opt/Wushi/


cd /path/to/Wushi/docker
./deploy.sh
